STL (Standard Template Library) in C++

Introduction: STL provides ready-to-use classes and functions for common data structures and algorithms, making coding faster and more efficient.

1. Vectors (Dynamic Arrays)

#include <vector>
using namespace std;

int main() {
  vector<int> v = {1, 2, 3};
  v.push_back(4);  // Adds 4 at the end

  for (int x : v) 
    cout << x << " ";
}

2. Lists (Doubly Linked List)

#include <list>
using namespace std;

int main() {
  list<int> l = {10, 20, 30};
  l.push_front(5);  // Adds at the beginning
  l.push_back(40);  // Adds at the end

  for (int x : l)
    cout << x << " ";
}

3. Sets (Unique & Sorted)

#include <set>
using namespace std;

int main() {
  set<int> s = {5, 3, 9, 1};
  s.insert(3);  // Duplicate, won't be added

  for (int x : s)
    cout << x << " ";  // Output: 1 3 5 9 (sorted)
}

4. Maps (Key-Value Pairs)

#include <map>
using namespace std;

int main() {
  map<string, int> age;
  age["Alice"] = 25;
  age["Bob"] = 30;

  for (auto p : age)
    cout << p.first << ": " << p.second << endl;
}

5. Stack (LIFO)

#include <stack>
using namespace std;

int main() {
  stack<int> s;
  s.push(1);
  s.push(2);
  cout << s.top();  // 2
  s.pop();
  cout << s.top();  // 1
}

6. Queue (FIFO)

#include <queue>
using namespace std;

int main() {
  queue<int> q;
  q.push(10);
  q.push(20);
  cout << q.front();  // 10
  q.pop();
  cout << q.front();  // 20
}

7. Iterators

#include <vector>
using namespace std;

int main() {
  vector<int> v = {1, 2, 3};
  vector<int>::iterator it;

  for (it = v.begin(); it != v.end(); ++it)
    cout << *it << " ";
}

8. Algorithms

STL provides built-in algorithms in <algorithm>.

Sort, Find, Count:

#include <algorithm>
#include <vector>
using namespace std;

int main() {
  vector<int> v = {5, 2, 8, 1};

  sort(v.begin(), v.end());  // Sort ascending

  if (find(v.begin(), v.end(), 2) != v.end())
    cout << "Found 2";

  int c = count(v.begin(), v.end(), 8);
  cout << "Count of 8: " << c;
}

9. Pairs & Tuples

Pair Example:

#include <utility>
using namespace std;

int main() {
  pair<string, int> p = {"Alice", 25};
  cout << p.first << " " << p.second;
}

Tuple Example:

#include <tuple>
using namespace std;

int main() {
  tuple<string, int, double> t = make_tuple("Bob", 30, 99.5);
  cout << get<0>(t) << " " << get<1>(t);
}

10. Real-World Example: Word Frequency Counter

#include <map>
#include <string>
#include <iostream>
using namespace std;

int main() {
  map<string, int> freq;
  string words[] = {"apple", "banana", "apple", "orange", "banana"};

  for (string word : words)
    freq[word]++;

  for (auto p : freq)
    cout << p.first << ": " << p.second << endl;
}
